[docs]def sort(seq):
    """
    Tool function for normal order, should not be used separately

    Args:
        seq (list): array

    Returns:
        list: integer e.g. swapped array, number of swaps
    """
    swap_counter = 0
    changed = True
    while changed:
        changed = False
        for i in range(len(seq) - 1):
            if seq[i] > seq[i + 1]:
                swap_counter += 1
                seq[i], seq[i + 1] = seq[i + 1], seq[i]
                changed = True

    return seq, swap_counter
